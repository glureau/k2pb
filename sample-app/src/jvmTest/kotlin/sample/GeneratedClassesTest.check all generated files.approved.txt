  override fun ProtobufReader.decode(protoCodec: DelegateProtoCodec): RequiredToOptionalStart? {
    var item: CommonClass? = null
    var b: String? = null

    while (!eof) {
      when (val tag = readTag()) {
        1 -> {
          readMessage {
            with(protoCodec) {
              decode(CommonClass::class)
            }
          }
          ?.let {
            item = it
          }
        }
        2 -> {
          b = readString()
        }
        else -> {
          // Ignore unknown tag, basic forward compatibility
          protoCodec.onUnknownProtoNumber(RequiredToOptionalStart::class, tag)
          skipElement()
        }
      }
    }
    return RequiredToOptionalStart(
      item = requireNotNull(item) { "Field 'item' is declared as not nullable but is null" },
      b = b ?: "",
    )
  }
